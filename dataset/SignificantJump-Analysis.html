import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.ticker as mticker


file_paths = {
    'v0': r"C:\Users\Soli1\Desktop\lsh_apg_metrics_audio_v0.csv",
    'v3': r"C:\Users\Soli1\Desktop\lsh_apg_metrics_audio_v3.csv",
    'v4': r"C:\Users\Soli1\Desktop\lsh_apg_metrics_audio_v1.csv"
}

data = {}
for version, path in file_paths.items():
    df = pd.read_csv(path, delimiter="\t", engine="python", header=None)
    df.columns = ['Raw']
    pattern = r"Candidate ID=(\d+).*?InsertionTime=([\d.]+)ms.*?Distance=([\d.e+-]+).*?Local Density=(\d+)"
    df[['Candidate_ID', 'InsertionTime', 'Distance', 'Local_Density']] = df['Raw'].str.extract(pattern)
    df['InsertionTime'] = pd.to_numeric(df['InsertionTime'], errors='coerce')
    df.dropna(subset=['InsertionTime'], inplace=True)
    data[version] = df

plt.figure(figsize=(10, 6))
for version, df in data.items():
    plt.plot(range(len(df)), df['InsertionTime'].cumsum(), label=f'{version} Cumulative Time')

plt.xlabel('Number of Nodes')
plt.ylabel('Cumulative Indexing Time (ms)')
plt.title('Comparison of Cumulative Indexing Time Across Versions')
plt.legend()
plt.grid(True)
plt.gca().xaxis.set_major_locator(mticker.MaxNLocator(integer=True)) 
plt.show()

df['InsertionTime_MA'] = df['InsertionTime'].rolling(window=5).mean()  # Adjust window size as necessary


df['Change_In_Time'] = df['InsertionTime_MA'].diff()
threshold = df['Change_In_Time'].std()  

# Identify where the changes are greater than the threshold
jumps = df[abs(df['Change_In_Time']) > threshold]


plt.figure(figsize=(14, 7))
plt.plot(df.index, df['InsertionTime'], 'b-', label='Actual Insertion Time', alpha=0.5)
plt.plot(df.index, df['InsertionTime_MA'], 'r-', label='Moving Average of Insertion Time')
