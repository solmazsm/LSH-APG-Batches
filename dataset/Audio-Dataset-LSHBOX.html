import struct

data_type_bytes = 4  # 4 bytes for float (float32)
num_vectors = 5  # Example: 5 vectors
vector_dim = 10  

# Generate random float32 vector data
vector_data = np.random.rand(num_vectors, vector_dim).astype(np.float32)
binary_data = struct.pack('iii', data_type_bytes, num_vectors, vector_dim)  
for vec in vector_data:
    binary_data += struct.pack(f'{vector_dim}f', *vec)  # Packing float32 vectors

unpacked_header = struct.unpack('iii', binary_data[:12])  # Extract header info
unpacked_vectors = np.array(struct.unpack(f'{num_vectors * vector_dim}f', binary_data[12:])).reshape(num_vectors, vector_dim)

plt.figure(figsize=(8, 4))
plt.imshow(unpacked_vectors, cmap='viridis', aspect='auto')
plt.colorbar(label="Float Values")
plt.xlabel("Vector Dimensions")
plt.ylabel("Vector Index")
plt.title("Visualization of Stored Vectors in Binary Format")
plt.show()
