import pandas as pd
import matplotlib.pyplot as plt


file_path = r"C:\Users\Soli1\Desktop\lsh_apg_metrics_audio_v1.csv"


filtered_data_v4 = pd.read_csv(file_path, delimiter="\t", engine="python", header=None)
filtered_data_v4.columns = ['Raw']


filtered_data_v4['Raw'] = filtered_data_v4['Raw'].str.replace(r'[^\x00-\x7F]+', '', regex=True)

pattern_all = r"Candidate ID=(\d+).*?InsertionTime=([\d.]+)ms.*?Distance=([\d.e+-]+).*?Local Density=(\d+)"
pattern_no_insertion = r"Candidate ID=(\d+).*?Distance=([\d.e+-]+).*?Local Density=(\d+)"

filtered_data_v4[['Candidate_ID', 'InsertionTime', 'Distance', 'Local_Density']] = filtered_data_v4['Raw'].str.extract(pattern_all)
filtered_data_v4[['Candidate_ID_alt', 'Distance_alt', 'Local_Density_alt']] = filtered_data_v4['Raw'].str.extract(pattern_no_insertion)

filtered_data_v4['Candidate_ID'] = filtered_data_v4['Candidate_ID'].fillna(filtered_data_v4['Candidate_ID_alt'])
filtered_data_v4['Distance'] = filtered_data_v4['Distance'].fillna(filtered_data_v4['Distance_alt'])
filtered_data_v4['Local_Density'] = filtered_data_v4['Local_Density'].fillna(filtered_data_v4['Local_Density_alt'])

filtered_data_v4.drop(columns=['Candidate_ID_alt', 'Distance_alt', 'Local_Density_alt'], inplace=True)

filtered_data_v4['Candidate_ID'] = pd.to_numeric(filtered_data_v4['Candidate_ID'], errors='coerce')
filtered_data_v4['InsertionTime'] = pd.to_numeric(filtered_data_v4['InsertionTime'], errors='coerce')
filtered_data_v4['Distance'] = pd.to_numeric(filtered_data_v4['Distance'], errors='coerce')
filtered_data_v4['Local_Density'] = pd.to_numeric(filtered_data_v4['Local_Density'], errors='coerce')

filtered_data_v4.dropna(subset=['Candidate_ID', 'InsertionTime', 'Distance', 'Local_Density'], inplace=True)

df = pd.read_csv(r"C:\Users\Soli1\Desktop\lsh_apg_metrics_audio_cleaned.csv")

df.columns = [col.strip() for col in df.columns]
df["Candidate_ID"] = pd.to_numeric(df["Candidate_ID"], errors="coerce")
df["Local_Density"] = pd.to_numeric(df["Local_Density"], errors="coerce")
df["Distance"] = pd.to_numeric(df["Distance"], errors="coerce")
df["InsertionTime"] = pd.to_numeric(df["InsertionTime"], errors="coerce")

df.dropna(inplace=True)

df["Time_Difference"] = df["InsertionTime"].diff()

df = df.dropna(subset=["Time_Difference"])  # Remove NaN rows
if df["Time_Difference"].nunique() == 1:  # Ensure variation exists
    print("Time_Difference has constant values! OLS regression cannot run.")
else:
    
    jump_threshold = np.percentile(df["Time_Difference"].dropna(), 90)
    df["Jump"] = df["Time_Difference"] > jump_threshold

    
    df["Candidate_ID_Log"] = np.log1p(df["Candidate_ID"])

    
    X = df[["Candidate_ID_Log", "Local_Density", "Distance"]]
    y = df["Jump"].astype(int)  # Binary classification (Jump: 1, No Jump: 0)

   
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)

  
    log_reg = LogisticRegression()
    log_reg.fit(X_scaled, y)

    feature_importance = pd.DataFrame({"Feature": X.columns, "Coefficient": log_reg.coef_[0]})
    print("Regression Coefficients (Impact on Jumps):\n", feature_importance)

   
    if len(df) > len(X.columns):  
        # Run OLS Regression
        X_ols = sm.add_constant(X_scaled)  # Add constant for intercept
        ols_model = sm.OLS(df["Time_Difference"], X_ols).fit()
        print("\n OLS Regression Summary (Continuous Impact on Insertion Time Jumps):\n", ols_model.summary())
    else:
        print("Not enough observations for OLS regression. Try collecting more data.")
